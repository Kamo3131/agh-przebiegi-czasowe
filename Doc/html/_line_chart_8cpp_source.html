141void LineChart::Draw_multiple_lines(sf::RenderTarget& target, sf::Color color_of_lines) const
142{
143 std::size_t safe_start = (m_start < 1) ? 1 : m_start;
144 std::size_t safe_end = (m_end > m_data.size()) ? m_data.size() : m_end;
145 if(m_data.size() < 2) return;
146 sf::VertexArray segments(sf::PrimitiveType::Lines);
147 float window_height = target.getSize().y;
148 for(std::size_t i = safe_start; i < safe_end; i++)
149 {
150 const Timestamp& a = m_data[i-1];
151 const Timestamp& b = m_data[i];
152
153 float ax = (a.Get_time() - m_view_min_time) * m_scale_X + m_origin.x + m_padding;
154 float bx = (b.Get_time() - m_view_min_time) * m_scale_X + m_origin.x + m_padding;
155
156 //Scale_Y will be higher than scale_X as voltage is lower than time in most cases
157 float ay = window_height - m_origin.y - (a.Get_voltage() - m_data_min_Y) * m_scale_Y;
158 float by = window_height - m_origin.y - (b.Get_voltage() - m_data_min_Y) * m_scale_Y;
159
160 if((ax < (m_origin.x + m_padding) && bx < (m_origin.x + m_padding))
161 || (ax > (m_origin.x + m_padding + m_width) && bx > (m_origin.x + m_padding + m_width)))
162 {
163 continue;
164 } else
165 if(ax < (m_origin.x + m_padding) && bx > (m_origin.x + m_padding))
166 {
167 ax = m_origin.x + m_padding;
168 } else
169 if(ax < (m_origin.x + m_width) && bx > (m_origin.x + m_width)){
170 bx = m_origin.x + m_width;
171 }
172 sf::Vertex vertex;
173 vertex.position = sf::Vector2f(ax, ay);
174 vertex.color = color_of_lines;

303void LineChart::Set_zoom(float new_zoom)
304{
305 constexpr float min_zoom = 1.f;
306 constexpr float max_zoom = 16.f;
307 if (new_zoom < min_zoom || new_zoom > max_zoom) return;
308 if (m_data.empty()) return;
309 constexpr float higher_than_one_zoom_round_border = 1.010f;
310 constexpr float lower_than_one_zoom_round_border = 0.990f;
311 if (higher_than_one_zoom_round_border > new_zoom && lower_than_one_zoom_round_border < new_zoom) new_zoom = 1.0f;
312
313 float zoom_ratio = new_zoom / m_zoom;
314
315 float cursor_x = static_cast<float>(m_cursor_position.x);
316 float from_min_to_cursor_time = (cursor_x - m_origin.x - m_padding) / m_scale_X;
317 float from_cursor_to_max_time = m_time_span - from_min_to_cursor_time;
318 float new_time_from_min = from_min_to_cursor_time / zoom_ratio;
319 float new_time_from_cursor = from_cursor_to_max_time / zoom_ratio;
320 float new_time_span = new_time_from_min + new_time_from_cursor;
321 if ( new_time_from_min > from_min_to_cursor_time){
322 m_view_min_time -= new_time_from_min - from_min_to_cursor_time;
323 } else {
324 m_view_min_time += from_min_to_cursor_time - new_time_from_min;
325 }
326 if ( new_time_from_cursor > from_cursor_to_max_time){
327 m_view_max_time += new_time_from_cursor - from_cursor_to_max_time;
328 } else {
329 m_view_max_time -= from_cursor_to_max_time - new_time_from_cursor;
330 }
331 m_time_span = new_time_span;
332 m_zoom = new_zoom;
333 Update_geometry();
334}
